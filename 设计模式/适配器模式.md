#### 适配器模式

- 主要是为了解决不同的方法或者类之间兼容性的问题，使得接口最终能够统一使用。

```js
//需要被适配的
class Socket {
  output(){
    return "我是220v"
  }
}
//电源适配器
class PowerAdaptor {
  constructor(socket){
    this.socket = socket
  }
  charge(){
    return this.socket.output() + "转化为20v"
  }
}
let s = new Socket()
let adaptor = new PowerAdaptor(s)
adaptor.charge()
```

```js
//浏览器环境和node环境下做适配
axios({
  url: "http://localhost:8080",
  method: "get"
})
function xhr(){
  return new Promise((resolve, reject)=>{
    let xhr = new XMLHttpRequest()
    xhr.open(method,url,true)
    xhr.onReadyStateChange = function(){
      if(xhr.readyState === 4){
        if(xhr.status === 200){
          resolve(xhr.responsetext)
        }
      }
    }
    xhr.send()
  }) 
}
function http(){
  let http = require("http")
  return new Promise((resolve, reject)=>{
    const options = {}
    http.request(options, function(response){
      let chunks = []
      response.on("data", function(chunk){
        chunks.push(chunk)
      })
      response.on("end", function(){
        resolve(chunks.toString())
      })
    })
  })
}
function getDefaultAdaptor(){
  let adaptor
  if(XMLHttpRequest){
    adaptor = xhr
  }else if(process){
    adaptor = http
  }
}
function axios(config){
  let defaultAdaptor = getDefaultAdaptor()
  return defaultAdaptor(config)
}
```
